2. The PDF Logic (lib/generatePdf.ts)
We’ll create a clean function that captures your Blueprint component. Because your site is dark, we’ll add a "Print Mode" to ensure the PDF looks professional on white paper (or keep it dark if you want that "cyber-deck" look).

TypeScript
import jsPDF from 'jspdf';
import html2canvas from 'html2canvas';

export const downloadBlueprint = async (elementId: string, fileName: string) => {
  const element = document.getElementById(elementId);
  if (!element) return;

  // 1. Capture the element as a high-res image
  const canvas = await html2canvas(element, {
    scale: 2, // High DPI for crisp text
    backgroundColor: "#080A08", // Matches your Obsidian background
    logging: false,
    useCORS: true // Important for capturing the site screenshot
  });

  const imgData = canvas.toDataURL('image/png');
  
  // 2. Create PDF (A4 size)
  const pdf = new jsPDF({
    orientation: 'portrait',
    unit: 'px',
    format: [canvas.width, canvas.height]
  });

  // 3. Add the image to the PDF
  pdf.addImage(imgData, 'PNG', 0, 0, canvas.width, canvas.height);
  
  // 4. Save the file
  pdf.save(`${fileName}_Nodus_Blueprint.pdf`);
};
