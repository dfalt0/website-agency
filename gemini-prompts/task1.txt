1. The Backend Engine: app/api/scan/route.ts
This script does the "thinking." It fetches the site, looks at the "invisible" headers, and parses the HTML for specific patterns.

TypeScript
import { NextResponse } from "next/server";

// Detection Patterns
const SIGNATURES = {
  js_frameworks: [
    { name: "React", test: /react/i, selector: 'script[src*="react"]' },
    { name: "Vue.js", test: /vue/i, selector: 'script[src*="vue"]' },
    { name: "Alpine.js", test: /alpine/i, selector: 'script[src*="alpine"]' },
  ],
  web_frameworks: [
    { name: "Next.js", test: /_next\/static/i, header: "x-powered-by" },
    { name: "Nuxt.js", test: /_nuxt/i },
    { name: "Express", header: "x-powered-by", value: /express/i },
  ],
  cms: [
    { name: "WordPress", test: /wp-content|wp-includes/i, meta: "generator" },
    { name: "Shopify", test: /shopify/i, selector: 'link[href*="shopify"]' },
    { name: "Webflow", test: /webflow/i, meta: "generator" },
  ],
  ui_frameworks: [
    { name: "Tailwind CSS", test: /tailwind/i, selector: 'link[href*="tailwind"]' },
    { name: "Bootstrap", test: /bootstrap/i, selector: 'link[href*="bootstrap"]' },
    { name: "Framer Motion", test: /framer-motion/i },
  ],
  servers_cdn: [
    { name: "Cloudflare", header: "server", value: /cloudflare/i },
    { name: "Vercel", header: "x-vercel-id" },
    { name: "Netlify", header: "server", value: /netlify/i },
    { name: "AWS CloudFront", header: "x-cache", value: /cloudfront/i },
  ],
};

export async function POST(req: Request) {
  try {
    const { url } = await req.json();
    if (!url) return NextResponse.json({ error: "URL is required" }, { status: 400 });

    const targetUrl = url.startsWith("http") ? url : `https://${url}`;
    
    // 1. Fetch the site with a timeout
    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), 8000);

    const response = await fetch(targetUrl, {
      signal: controller.signal,
      headers: { "User-Agent": "Mozilla/5.0 (Nodus-Scanner/1.0)" },
    });

    const html = await response.text();
    const headers = Object.fromEntries(response.headers);
    clearTimeout(timeoutId);

    const stack: any[] = [];

    // 2. Run the Detection Logic
    Object.entries(SIGNATURES).forEach(([category, techs]) => {
      techs.forEach((tech: any) => {
        let detected = false;

        // Check Headers
        if (tech.header && headers[tech.header]) {
          if (!tech.value || tech.value.test(headers[tech.header])) detected = true;
        }
        // Check HTML content
        if (tech.test && tech.test.test(html)) detected = true;
        
        // Check Meta tags (simple string search for speed)
        if (tech.meta && html.includes(`content="${tech.name}`)) detected = true;

        if (detected) {
          stack.push({
            name: tech.name,
            type: category.replace("_", " ").toUpperCase(),
            confidence: "high"
          });
        }
      });
    });

    // 3. Fallback: If we find nothing, at least return the server header
    if (stack.length === 0 && headers["server"]) {
      stack.push({ name: headers["server"], type: "WEB SERVER", confidence: "medium" });
    }

    return NextResponse.json({ stack, status: "success", url: targetUrl });

  } catch (error: any) {
    return NextResponse.json({ error: "Site unreachable or blocking scans." }, { status: 500 });
  }
}
